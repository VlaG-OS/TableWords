# Пример файла для импорта Word Matcher
# Формат: слово = перевод

[Базовые слова]
hello = привет
goodbye = пока
thank you = спасибо
please = пожалуйста
yes = да
no = нет

[Числа]
one = один
two = два
three = три
four = четыре
five = пять
ten = десять

[Путешествия]
airport = аэропорт
hotel = отель
ticket = билет
passport = паспорт
train = поезд
bus = автобус
